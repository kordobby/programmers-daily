```javascript
function solution(sizes) {
  let [width, height] = [-1, -1];

  // 가로 길이의 최댓값, 세로 길이의 최댓값 구하기
  sizes.forEach(([x, y]) => {
    let w = Math.max(x, y);
    let h = Math.min(x, y);
    [width, height] = [w > width ? w : width, h > height ? h : height];
  });
  return width * height;
}
```
