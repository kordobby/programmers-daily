#### 첫 번째 풀이

```javascript
function solution(number, limit, power) {
  let knight = 0;
  const getPower = (value) => {
    let cnt = 1;
    for (i = 1; i <= value / 2; i++) {
      if (value % i === 0) {
        cnt++;
      }
    }
    knight += cnt > limit ? power : cnt;
  };

  for (k = 1; k <= number; k++) {
    getPower(k);
  }

  return knight;
}
```
